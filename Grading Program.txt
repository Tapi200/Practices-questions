#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <time.h>
using namespace std;

bool check_guess(int guess, int random) {
	if (guess == random) {
		cout << "You guessed the correct number" << endl;
		return true;
	}
	if (guess < random) {
		cout << "Guess too low" << endl;
		return false;
	}
	if (guess > random) {
		cout << "Guess too high" << endl;
		return false;
	}
}

int main() {

	int rNumber;
	int rGuess;
	int finished = false;
	int guesses = 0;

	srand((unsigned) time(0));

	rNumber = rand() % 100 + 1;

	cout << rNumber << endl;

	cout << "Guess the number" << endl;

	while (!finished) {
		cin >> rGuess;
		guesses += 1;
		finished = check_guess(rGuess, rNumber);
	}
	cout << "number of guesses:  " << guesses << endl;
	return 0;
}

#include <iostream>

int sumArray(int array[]){
	int sum = 0;
	
	unsigned length = sizeof(array) / sizeof(array[0]);

	for(int i = 0; i < length; i ++){
		sum = sum + array[i];
	}
	return sum;
}
